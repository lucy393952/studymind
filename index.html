function renderSubjects() {

    return;

  }

  el.innerHTML=subjects.map(s=>{

    const d=daysUntil(s.date);

    const bc=d>14?'d-ok':d>5?'d-warn':'d-urg';

    const bt=d<0?'Vorbei':d===0?'Heute!':d+' Tage';

    const pct=Math.min(100,s.progress||0);

    const tc=s.topics&&s.topics.length?`· ${s.topics.length} Thema(en)`:'';
